LODOP获取打印状态码含义两种方式

之前的博文：LODOP获取打印机状态码和状态码含义测试，中测试获取了状态码和状态码含义。
该文中获取状态码含义的方式是：LODOP.GET_VALUE("PRINT_STATUS_TEXT",JOB代码)
之前的博文中也获取了状态码，获取状态码的时候用的是job代码和PRINT_STATUS_TEXT，没有直接用到该状态码，前面的博文介绍通过状态码除以1024，取余数，可以得到合计值，然后根据合计值可以对照信息找到状态码含义，也可以通过如下方法解析生成的状态码，获取到状态码含义。
测试代码：
两种获取状态码含义的方法：
1.job代码和PRINT_STATUS_TEXT
2.根据job代码获取打印状态码，然后获取打印状态码，再获取状态码含义。

        var LODOP; //声明为全局变量 
    function prn1_preview() {//获取job代码
        LODOP=getLodop();  
        LODOP.PRINT_INIT("");
        LODOP.ADD_PRINT_TEXT(50,231,260,39,"一个文本");
        LODOP.SET_PRINT_MODE("CATCH_PRINT_STATUS",true);
        if (LODOP.CVERSION) {
            LODOP.On_Return=function(TaskID,Value){document.getElementById('T1').value=Value;};
            LODOP.PRINTA();
        } else document.getElementById('T1').value=LODOP.PRINTA();
    };
    function prn2_preview() {//获取打印状态码
          LODOP=getLodop(); 
        if (LODOP.CVERSION) CLODOP.On_Return=function(TaskID,Value){document.getElementById('T2').value=Value;}; 
        var stResult=LODOP.GET_VALUE("PRINT_STATUS_ID",document.getElementById('T1').value);
        if (!LODOP.CVERSION) document.getElementById('T2').value=stResult; 
    };
    function prn3_preview() {//获取该打印状态码的含义
          LODOP=getLodop(); 
        if (LODOP.CVERSION) CLODOP.On_Return=function(TaskID,Value){document.getElementById('T3').value=Value;}; 
        var stResult=LODOP.GET_VALUE("PRINT_STATUS_TEXT",document.getElementById('T1').value);
        if (!LODOP.CVERSION) document.getElementById('T3').value=stResult; 
    };
    function prn4_preview() {//获取该打印状态码的含义
     var sd=document.getElementById('T2').value;
      getStatuMessage(sd);
    };
    function getStatuMessage(statusID) { 
        var messages="";
        if (statusID & 1) messages += "已暂停 -";
        if (statusID & 2) messages += "错误 -";
        if (statusID & 4) messages += "正删除 -";
        if (statusID & 8) messages += "进入队列 -";
        if (statusID & 16) messages += "正在打印 -";
        if (statusID & 32) messages += "脱机 -";
        if (statusID & 64) messages += "缺纸 -";
        if (statusID & 128) messages += "打印结束 -";
        if (statusID & 256) messages += "已删除 -";
        if (statusID & 512) messages += "堵了 -";
        if (statusID & 1024) messages += "用户介入 -";
        if (statusID & 2048) messages += "正在重新启动 -";
        document.getElementById('T4').value=messages; 
    };

附加测试：
测试1：测试了下打印机队列本身有其他任务处在错误-正在打印状态，然后再发送打印任务到队列，测试结果是状态码是0，不能从状态码得到状态含义，但是直接用PRINT_STATUS_TEXT这种方法，可以获取到正在进入队列的状态码。
实际上个这个任务是在打印机队列进行排队的。

测试2：打印后，删除错误-正在大打印状态的额任务，可获取到删除状态。
